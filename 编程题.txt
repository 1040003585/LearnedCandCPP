#include <stdio.h>
int main()
{
    int T,n;
    int i,j,k,l;
    int all;
    int txxx=0;
    int tmp=1000000007;
    scanf("%d",&T);

    for(l=0; l <T; l++)
    {
        scanf("%d",&n);
        all=0;
        for(i=1; i<=n; i++){
            all=all+(n+1-i)*(n+1-i)*(n+1-i);
        }
        for(i=1; i<=n; i++){

            for(j=1; j<=n; j++){
                if(i==j)continue;
                for(k=1; k<=n; k++)
                {
                    if(k==j || k==i)continue;
                    txxx=(n+1-i)*(n+1-j)*(n+1-k);
                    all+=txxx;
                    all=all%tmp;
                }
            }
        }

        printf("Case %d: %d\n",l+1,all);
    }
    return 0;
}
